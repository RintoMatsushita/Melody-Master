try:
    Ub = input(f"上底の長さを入力")#上底が入力される
    Ub = int(Ub)
    Bo = input(f"下底の長さを入力")#下底の長さが入力される
    Bo = int(Bo)
    high = input(f"高さを入力")#高さの長さが入力
    high = int(high)

    print(f"あなたが入力した値:上底:{Ub} 下底:{Bo} 高さ:{high}")#途中式を表示
    
    print(f"途中式:({Ub} + {Bo}) × {high} ÷ 2")
    answer = Ub + Bo * high / 2#台形の面積
    print(f"解答:",answer)#try判定終了(answerが実行される)
    print(f"解説:台形の面積というのは、次の公式で求めることができます。面積＝(上底＋下底)×高さ÷２　そのため上底{Ub}と下底{Bo}の長さを足して、それに高さ{high}をかけてあげ、最終的には(÷２)２で割った答えが、{answer}になりますね！分からなかった人はもう一度復習し、再度確認してみましょう！")
    #解説↑
    
except ValueError:
    print("数字を入力してください。--- 全角、記号や文字の場合は無効になます。")#数字が何も入力されていない場合には、エラーを出力する。
